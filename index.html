# Sample DMARC report parsing
def parse_dmarc_report(file):
    try:
        root = ET.fromstring(file.read())
        report = {
            'org': safe_find_text(root, './/org_name'),
            'email': safe_find_text(root, './/email'),
            'report_id': safe_find_text(root, './/report_id'),
            'date_range': {
                'start': timestamp_to_date(root.find('.//date_range/begin').text),
                'end': timestamp_to_date(root.find('.//date_range/end').text)
            },
            'records': [parse_record(record) for record in root.findall('.//record')]
        }
        return report
    except Exception as e:
        return {'error': str(e)}
